## Overview

The Physics System is the main simulation engine in JoltC. It manages bodies, constraints, collision detection, and simulation stepping.

## Structures

### JPH_PhysicsSystemSettings

Configuration settings for creating a physics system.

```c
typedef struct JPH_PhysicsSystemSettings {
    uint32_t maxBodies;
    uint32_t numBodyMutexes;
    uint32_t maxBodyPairs;
    uint32_t maxContactConstraints;
    uint32_t _padding;
    JPH_BroadPhaseLayerInterface* broadPhaseLayerInterface;
    JPH_ObjectLayerPairFilter* objectLayerPairFilter;
    JPH_ObjectVsBroadPhaseLayerFilter* objectVsBroadPhaseLayerFilter;
} JPH_PhysicsSystemSettings;
```

<ParamField path="maxBodies" type="uint32_t">
  Maximum number of bodies that can be added to the system. Default: 10240.
</ParamField>

<ParamField path="numBodyMutexes" type="uint32_t">
  Number of mutexes to allocate for body access. Use 0 for auto-calculation. Default: 0.
</ParamField>

<ParamField path="maxBodyPairs" type="uint32_t">
  Maximum number of body pairs that can be in contact. Default: 65536.
</ParamField>

<ParamField path="maxContactConstraints" type="uint32_t">
  Maximum number of contact constraints. Default: 10240.
</ParamField>

<ParamField path="broadPhaseLayerInterface" type="JPH_BroadPhaseLayerInterface*">
  Broad phase layer interface for collision filtering.
</ParamField>

<ParamField path="objectLayerPairFilter" type="JPH_ObjectLayerPairFilter*">
  Filter to determine which object layer pairs can collide.
</ParamField>

<ParamField path="objectVsBroadPhaseLayerFilter" type="JPH_ObjectVsBroadPhaseLayerFilter*">
  Filter to determine which object layers can collide with broad phase layers.
</ParamField>

### JPH_PhysicsSettings

Runtime physics simulation settings.

```c
typedef struct JPH_PhysicsSettings {
    int maxInFlightBodyPairs;
    int stepListenersBatchSize;
    int stepListenerBatchesPerJob;
    float baumgarte;
    float speculativeContactDistance;
    float penetrationSlop;
    float linearCastThreshold;
    float linearCastMaxPenetration;
    float manifoldTolerance;
    float maxPenetrationDistance;
    float bodyPairCacheMaxDeltaPositionSq;
    float bodyPairCacheCosMaxDeltaRotationDiv2;
    float contactNormalCosMaxDeltaRotation;
    float contactPointPreserveLambdaMaxDistSq;
    uint32_t numVelocitySteps;
    uint32_t numPositionSteps;
    float minVelocityForRestitution;
    float timeBeforeSleep;
    float pointVelocitySleepThreshold;
    bool deterministicSimulation;
    bool constraintWarmStart;
    bool useBodyPairContactCache;
    bool useManifoldReduction;
    bool useLargeIslandSplitter;
    bool allowSleeping;
    bool checkActiveEdges;
} JPH_PhysicsSettings;
```

<ParamField path="numVelocitySteps" type="uint32_t">
  Number of solver velocity iterations per simulation step.
</ParamField>

<ParamField path="numPositionSteps" type="uint32_t">
  Number of solver position iterations per simulation step.
</ParamField>

<ParamField path="baumgarte" type="float">
  Baumgarte stabilization factor for position correction.
</ParamField>

<ParamField path="speculativeContactDistance" type="float">
  Distance threshold for speculative contact generation (in meters).
</ParamField>

<ParamField path="penetrationSlop" type="float">
  Allowed penetration before position correction kicks in (in meters).
</ParamField>

<ParamField path="minVelocityForRestitution" type="float">
  Minimum velocity for restitution (bounce) to be applied.
</ParamField>

<ParamField path="timeBeforeSleep" type="float">
  Time a body must be below sleep thresholds before going to sleep (in seconds).
</ParamField>

<ParamField path="pointVelocitySleepThreshold" type="float">
  Velocity threshold below which a body can go to sleep.
</ParamField>

<ParamField path="deterministicSimulation" type="bool">
  Enable deterministic simulation (same inputs produce same outputs).
</ParamField>

<ParamField path="allowSleeping" type="bool">
  Allow bodies to go to sleep when at rest.
</ParamField>

<ParamField path="checkActiveEdges" type="bool">
  Check for active edges during collision detection.
</ParamField>

## Example

```c
#include <joltc.h>
#include <stdio.h>

int main() {
    if (!JPH_Init()) {
        return 1;
    }
    
    // Set up collision layers (simplified example)
    JPH_BroadPhaseLayerInterface* bpInterface = 
        JPH_BroadPhaseLayerInterfaceTable_Create(2, 2);
    JPH_ObjectLayerPairFilter* layerFilter = 
        JPH_ObjectLayerPairFilterTable_Create(2);
    JPH_ObjectVsBroadPhaseLayerFilter* bpFilter = 
        JPH_ObjectVsBroadPhaseLayerFilterTable_Create(bpInterface, 2, layerFilter, 2);
    
    // Create physics system
    JPH_PhysicsSystemSettings settings = {
        .maxBodies = 1024,
        .numBodyMutexes = 0,
        .maxBodyPairs = 1024,
        .maxContactConstraints = 1024,
        .broadPhaseLayerInterface = bpInterface,
        .objectLayerPairFilter = layerFilter,
        .objectVsBroadPhaseLayerFilter = bpFilter
    };
    
    JPH_PhysicsSystem* system = JPH_PhysicsSystem_Create(&settings);
    if (!system) {
        fprintf(stderr, "Failed to create physics system\n");
        JPH_Shutdown();
        return 1;
    }
    
    // Set gravity
    JPH_Vec3 gravity = { 0.0f, -9.81f, 0.0f };
    JPH_PhysicsSystem_SetGravity(system, &gravity);
    
    // Create job system
    JobSystemThreadPoolConfig jobConfig = {
        .maxJobs = 2048,
        .maxBarriers = 8,
        .numThreads = -1  // Auto-detect CPU count
    };
    JPH_JobSystem* jobSystem = JPH_JobSystemThreadPool_Create(&jobConfig);
    
    // Simulation loop
    const float deltaTime = 1.0f / 60.0f;
    for (int i = 0; i < 600; i++) {
        JPH_PhysicsUpdateError error = JPH_PhysicsSystem_Update(
            system, deltaTime, 1, jobSystem
        );
        
        if (error != JPH_PhysicsUpdateError_None) {
            fprintf(stderr, "Physics update error: 0x%x\n", error);
        }
    }
    
    // Clean up
    JPH_JobSystem_Destroy(jobSystem);
    JPH_PhysicsSystem_Destroy(system);
    JPH_Shutdown();
    
    return 0;
}
```
