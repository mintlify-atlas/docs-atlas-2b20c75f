## Overview

Collision filters control which objects can collide with each other in JoltC. The system uses a two-tier filtering approach:

1. **Broad Phase Layers** - Coarse filtering for performance
2. **Object Layers** - Fine-grained collision control

## Broad Phase Layer Interface

The broad phase layer interface maps object layers to broad phase layers.

</ParamField>

</ResponseField>

## Custom Filters

You can create custom filters for queries using callback functions.

### Broad Phase Layer Filter

<CodeGroup>
```c Create Custom Filter
typedef struct JPH_BroadPhaseLayerFilter_Procs {
    bool (*ShouldCollide)(void* userData, JPH_BroadPhaseLayer layer);
} JPH_BroadPhaseLayerFilter_Procs;

void JPH_BroadPhaseLayerFilter_SetProcs(
    const JPH_BroadPhaseLayerFilter_Procs* procs
);

JPH_BroadPhaseLayerFilter* JPH_BroadPhaseLayerFilter_Create(void* userData);
void JPH_BroadPhaseLayerFilter_Destroy(JPH_BroadPhaseLayerFilter* filter);
```
</CodeGroup>

<ParamField path="ShouldCollide" type="callback">
  Called to determine if a broad phase layer should be tested.
  
  **Parameters:**
  - `userData` - User data passed during filter creation
  - `layer` - The broad phase layer to test
  
  **Returns:** `true` if the layer should be tested
</ParamField>

</CodeGroup>

<ParamField path="ShouldCollide" type="callback">

</ParamField>

### Body Filter

<CodeGroup>
```c Create Custom Filter
typedef struct JPH_BodyFilter_Procs {
    bool (*ShouldCollide)(void* userData, JPH_BodyID bodyID);
    bool (*ShouldCollideLocked)(void* userData, const JPH_Body* body);
} JPH_BodyFilter_Procs;

void JPH_BodyFilter_SetProcs(const JPH_BodyFilter_Procs* procs);

JPH_BodyFilter* JPH_BodyFilter_Create(void* userData);
void JPH_BodyFilter_Destroy(JPH_BodyFilter* filter);
```
</CodeGroup>

<ParamField path="ShouldCollide" type="callback">
  Called when body is not locked. Should return quickly.
  
  **Parameters:**
  - `userData` - User data
  - `bodyID` - The body ID to test
</ParamField>

<ParamField path="ShouldCollideLocked" type="callback">
  Called when body is locked. Can access full body information.
  
  **Parameters:**
  - `userData` - User data
  - `body` - The locked body to test
</ParamField>

## Example Usage

<CodeGroup>
```c Table-Based Filtering
// Define layers
#define LAYER_STATIC 0
#define LAYER_MOVING 1
#define LAYER_DEBRIS 2

#define BP_LAYER_NON_MOVING 0
#define BP_LAYER_MOVING 1

// Create broad phase interface
JPH_BroadPhaseLayerInterface* bp_interface =
    JPH_BroadPhaseLayerInterfaceTable_Create(3, 2);

JPH_BroadPhaseLayerInterfaceTable_MapObjectToBroadPhaseLayer(
    bp_interface, LAYER_STATIC, BP_LAYER_NON_MOVING);
JPH_BroadPhaseLayerInterfaceTable_MapObjectToBroadPhaseLayer(
    bp_interface, LAYER_MOVING, BP_LAYER_MOVING);
JPH_BroadPhaseLayerInterfaceTable_MapObjectToBroadPhaseLayer(
    bp_interface, LAYER_DEBRIS, BP_LAYER_MOVING);

// Create object layer filter
JPH_ObjectLayerPairFilter* obj_filter =
    JPH_ObjectLayerPairFilterTable_Create(3);

// Static doesn't collide with static
JPH_ObjectLayerPairFilterTable_DisableCollision(
    obj_filter, LAYER_STATIC, LAYER_STATIC);

// Everything else collides
JPH_ObjectLayerPairFilterTable_EnableCollision(
    obj_filter, LAYER_STATIC, LAYER_MOVING);
JPH_ObjectLayerPairFilterTable_EnableCollision(
    obj_filter, LAYER_STATIC, LAYER_DEBRIS);
JPH_ObjectLayerPairFilterTable_EnableCollision(
    obj_filter, LAYER_MOVING, LAYER_MOVING);
JPH_ObjectLayerPairFilterTable_EnableCollision(
    obj_filter, LAYER_MOVING, LAYER_DEBRIS);
JPH_ObjectLayerPairFilterTable_EnableCollision(
    obj_filter, LAYER_DEBRIS, LAYER_DEBRIS);

// Create object vs broad phase filter
JPH_ObjectVsBroadPhaseLayerFilter* obj_vs_bp_filter =
    JPH_ObjectVsBroadPhaseLayerFilterTable_Create(
        bp_interface, 2, obj_filter, 3);
```

```c Mask-Based Filtering
// Define groups
#define GROUP_STATIC 0
#define GROUP_DYNAMIC 1
#define GROUP_SENSOR 2

// Create mask-based filters
JPH_ObjectLayerPairFilter* obj_filter =
    JPH_ObjectLayerPairFilterMask_Create();

// Create layers with collision masks
JPH_ObjectLayer static_layer =
    JPH_ObjectLayerPairFilterMask_GetObjectLayer(
        GROUP_STATIC,
        (1 << GROUP_DYNAMIC) | (1 << GROUP_SENSOR)
    );

JPH_ObjectLayer dynamic_layer =
    JPH_ObjectLayerPairFilterMask_GetObjectLayer(
        GROUP_DYNAMIC,
        (1 << GROUP_STATIC) | (1 << GROUP_DYNAMIC)
    );

JPH_ObjectLayer sensor_layer =
    JPH_ObjectLayerPairFilterMask_GetObjectLayer(
        GROUP_SENSOR,
        (1 << GROUP_STATIC) | (1 << GROUP_DYNAMIC)
    );
```

```c Custom Body Filter
bool my_body_filter(void* userData, JPH_BodyID bodyID) {
    // Filter out specific bodies
    JPH_BodyID* ignore_id = (JPH_BodyID*)userData;
    return bodyID != *ignore_id;
}

JPH_BodyFilter_Procs procs = {
    .ShouldCollide = my_body_filter,
    .ShouldCollideLocked = NULL
};

JPH_BodyFilter_SetProcs(&procs);

JPH_BodyID ignore_body = 42;
JPH_BodyFilter* filter = JPH_BodyFilter_Create(&ignore_body);

// Use in raycast
JPH_RayCastResult hit;
JPH_NarrowPhaseQuery_CastRay(query, &origin, &direction, &hit,
    NULL, NULL, filter);

JPH_BodyFilter_Destroy(filter);
```
</CodeGroup>
